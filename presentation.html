Anatomy of an Electron app

Just 3 required files:

  1. package.json
  2. Main Process (index.js)
  3. Renderer Process (index.html)

package.json

                    
{
  "name": "electron-bare-bones",
  "version": "0.1.0",
  "main": "index.js"
}

Main Process (index.js)

                    
const electron = require('electron')

let win // global ref so window is not closed with JS is garbage collected

electron.app.on('ready', () => {
  win = new electron.BrowserWindow({ width: 800, height: 600 })
  win.loadURL(`file://${__dirname}/index.html`)
})

Renderer Process (index.html)

                    
<!DOCTYPE html>
<html>
  <head>
    <title>Electron Bare Bones App</title>
  </head>

  <body>
    <h1>Hello World</h1>

    <script>
      alert('Hi!')
    </script>
  </body>
</html>
